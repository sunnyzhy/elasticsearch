# 新增字段
```
POST /${index}/_mapping
{
  "properties": {
    "${field}": {
	"type": "keyword",
	"index": true
    }
  }
}
```

# 分组(group by)
```
POST /${index}/_search
{
  "query": {
    "bool": {
      "filter": [
        {"term": {
          "${field1}": "${field1_value}"
        }},
        {"term": {
          "${field2}": "${field2_value}"
        }}
        ]
    }
  }, 
  "from": 0, 
  "size": 0,
  "aggs": {
    "${alias}": {
      "terms": {
        "field": "${field}"
      }
    }
  }
}
```

# 修改字段的值
```
POST /${index}/_update/${id}
{
  "doc": {
    "${field}": "${field_value}"
  }
}
```
